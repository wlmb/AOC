#+begin_src perl :tangle 1a.pl :shebang #!/usr/bin/env perl :results output
use v5.40;
use PDL;
use PDL::NiceSlice;
my @arr;
while(<>){
    chomp;
    push @arr, [split ' '];
}
say +(pdl([@arr])->mv(1,0)->inplace->qsort->mv(0,1)*pdl(1,-1))->sumover->abs->sumover;

#+end_src
